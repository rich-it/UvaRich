#include<bits/stdc++.h>

using namespace std;

int oa(int n) // the algorithm mentioned in problem
{
    int coun=0;
    while(1)
    {
        coun++; //this is the cycle nymber
        if(n==1)
            return coun;

        else if(n%2==0)
            n=n/2 ;
        else
            n=n*3+1;
    }
}

int main()
{
    int n1,n2,m=0,rv,x,y;


    while(scanf("%d %d",&n1,&n2)==2){
    if(n1<n2) {     //we have to check if i>j or i<j cz in problem it didn't i will always < j
            x=n1;
            y=n2;
    }
    else {
            x=n2;
            y=n1;
    }
    for(int i=x; i<=y; i++)
    {
        rv= oa(i); //calling the algorithm funtion to get cycle numbers
        if(rv>m)
            m=rv; //for getting highest cycle number
    }
    printf("%d %d %d\n",n1,n2,m);
    m=0; //for every time the loop is done with the calculation the final result should be 0 and the program will rework
}
}
